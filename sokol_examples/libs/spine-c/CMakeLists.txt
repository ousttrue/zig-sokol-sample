fips_begin_lib(spine-c)
    fips_src(src/spine)
    fips_src(include/spine)
    target_include_directories(spine-c PUBLIC include)
    if (FIPS_CLANG OR FIPS_GCC)
        target_compile_options(spine-c PRIVATE -Wno-sign-conversion -Wno-unused-but-set-variable -Wno-implicit-fallthrough -Wno-parentheses)
    endif()
    if (FIPS_CLANG)
        target_compile_options(spine-c PRIVATE -Wno-shorten-64-to-32)
    endif()
    if (FIPS_MSVC)
        target_compile_options(spine-c PRIVATE /wd4146 /wd4244 /wd4267)
    endif()
fips_end_lib()
